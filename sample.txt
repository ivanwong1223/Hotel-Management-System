public class Customer {
    private String name;
    private String contact;
    private String email;
    private String ic;
    private String checkinDate;
    private String checkoutDate;
    private String checkinMonth;
    private String checkoutMonth;
    private String year;
    private String roomSelected;
    
    public Customer (String name, String contact, String email, String ic, String checkinDate, String checkoutDate, String checkinMonth, String checkoutMonth, String year, String roomSelected) {
        this.name = name;
        this.contact = contact;
        this.email = email;
        this.ic = ic;
        this.checkinDate = checkinDate;
        this.checkoutDate = checkoutDate;
        this.checkinMonth = checkinMonth;
        this.checkoutMonth = checkoutMonth;
        this.year = year;
        this.roomSelected = roomSelected;
    }
    
    
    public String getName() {
        if (name == null || name.isEmpty()) {
            throw new IllegalArgumentException("Name cannot be null or empty");
        }
        if (!name.matches("[a-zA-Z]+")) {
            throw new IllegalArgumentException("Name must contain only letters");
        }
        return name;
    }

    public String getContact() {
        return contact;
    }

    public String getEmail() {
        return email;
    }

    public String getIc() {
        return ic;
    }

    public String getCheckinDate() {
        return checkinDate;
    }

    public String getCheckoutDate() {
        return checkoutDate;
    }

    public String getCheckinMonth() {
        return checkinMonth;
    }

    public String getCheckoutMonth() {
        return checkoutMonth;
    }

    public String getYear() {
        return year;
    }
    
    public String getRoomSelection(){
        return roomSelected;
    }
    
}

this is the sub class bookingDate, it inherits to the Customer class.
public class bookingDate extends Customer {
    private String dobin;
    private String dobout;
    
    
    public bookingDate(String name, String contact, String email, String ic, String checkinDate, String checkoutDate, String checkinMonth, String checkoutMonth, String year, String dobin, String dobout, String roomSelected){
        super(name, contact, email, ic, checkinDate, checkoutDate, checkinMonth, checkoutMonth, year, roomSelected);
        this.dobin = dobin;
        this.dobout = dobout;
    }
    
    public String getDobin() {
        this.dobin = getCheckinDate() + "-" + getCheckinMonth() + "-" + getYear();
        return dobin;
    }

    public String getDobout() {
        this.dobout = getCheckoutDate() + "-" + getCheckoutMonth() + "-" + getYear();
        return dobout;
    }
    
}
